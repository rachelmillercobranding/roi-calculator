<script>


const clients = document.getElementById("clientsSlider")

const rate = document.getElementById("rateSlider")


const clientsValue = document.getElementById("clientsValue")

const rateValue = document.getElementById("rateValue")


const monthly = document.getElementById("monthly")

const annual = document.getElementById("annual")

const roiText = document.getElementById("roiText")


function calculate(){


const weeklyClients = clients.value

const sessionRate = rate.value


clientsValue.innerText = weeklyClients

rateValue.innerText = "$" + sessionRate


const monthlyIncome = weeklyClients * sessionRate * 4.3

const yearlyIncome = monthlyIncome * 12


monthly.innerText = "$" + Math.round(monthlyIncome).toLocaleString()

annual.innerText = "$" + Math.round(yearlyIncome).toLocaleString()


const baseline = 100

const increase = sessionRate - baseline


if(increase > 0){

const extra = weeklyClients * increase * 4.3 * 12

roiText.innerText = "Increasing your rate from $100 to $" + sessionRate + " adds about $" + Math.round(extra).toLocaleString() + " per year."

}else{

roiText.innerText = ""

}


}


clients.oninput = calculate

rate.oninput = calculate


calculate()


</script>
